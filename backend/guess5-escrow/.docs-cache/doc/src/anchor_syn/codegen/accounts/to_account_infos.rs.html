5// Generates the `ToAccountInfos` trait implementation.
6pub fn generate(accs: &AccountsStruct) -> proc_macro2::TokenStream {
7    let name = &accs.ident;
8    let ParsedGenerics {
9        combined_generics,
10        trait_generics,
11        struct_generics,
12        where_clause,
13    } = generics(accs);
14
15    let to_acc_infos: Vec<proc_macro2::TokenStream> = accs
16        .fields
17        .iter()
18        .map(|f: &AccountField| {
19            let name = &f.ident();
20            quote! { account_infos.extend(self.#name.to_account_infos()); }
21        })
22        .collect();
23    quote! {
24        #[automatically_derived]
25        impl<#combined_generics> anchor_lang::ToAccountInfos<#trait_generics> for #name <#struct_generics> #where_clause{
26            fn to_account_infos(&self) -> Vec<anchor_lang::solana_program::account_info::AccountInfo<#trait_generics>> {
27                let mut account_infos = vec![];
28
29                #(#to_acc_infos)*
30
31                account_infos
32            }
33        }
34    }
