6pub fn generate(program: &Program) -> proc_macro2::TokenStream {
7    // Generate cpi methods for global methods.
8    let global_cpi_methods: Vec<proc_macro2::TokenStream> = program
9        .ixs
10        .iter()
11        .map(|ix| {
12            let accounts_ident: proc_macro2::TokenStream = format!("crate::cpi::accounts::{}", &ix.anchor_ident.to_string()).parse().unwrap();
13            let cpi_method = {
14                let name = &ix.raw_method.sig.ident;
15                let name_str = name.to_string();
16                let ix_variant = generate_ix_variant(&name_str, &ix.args);
17                let method_name = &ix.ident;
18                let args: Vec<&syn::PatType> = ix.args.iter().map(|arg| &arg.raw_arg).collect();
19                let discriminator = {
20                    let name = generate_ix_variant_name(&name_str);
21                    quote! { <instruction::#name as anchor_lang::Discriminator>::DISCRIMINATOR }
22                };
23                let ret_type = &ix.returns.ty.to_token_stream();
24                let ix_cfgs = &ix.cfgs;
25                let (method_ret, maybe_return) = match ret_type.to_string().as_str() {
26                    "()" => (quote! {anchor_lang::Result<()> }, quote! { Ok(()) }),
27                    _ => (
28                        quote! { anchor_lang::Result<crate::cpi::Return::<#ret_type>> },
29                        quote! { Ok(crate::cpi::Return::<#ret_type> { phantom: crate::cpi::PhantomData }) }
30                    )
31                };
32
33                quote! {
34                    #(#ix_cfgs)*
35                    pub fn #method_name<'a, 'b, 'c, 'info>(
36                        ctx: anchor_lang::context::CpiContext<'a, 'b, 'c, 'info, #accounts_ident<'info>>,
37                        #(#args),*
38                    ) -> #method_ret {
39                        let ix = {
40                            let ix = instruction::#ix_variant;
41                            let mut data = Vec::with_capacity(256);
42                            data.extend_from_slice(#discriminator);
43                            AnchorSerialize::serialize(&ix, &mut data)
44                                .map_err(|_| anchor_lang::error::ErrorCode::InstructionDidNotSerialize)?;
45                            let accounts = ctx.to_account_metas(None);
46                            anchor_lang::solana_program::instruction::Instruction {
47                                program_id: ctx.program.key(),
48                                accounts,
49                                data,
50                            }
51                        };
52                        let mut acc_infos = ctx.to_account_infos();
53                        anchor_lang::solana_program::program::invoke_signed(
54                            &ix,
55                            &acc_infos,
56                            ctx.signer_seeds,
57                        ).map_or_else(
58                            |e| Err(Into::into(e)),
59                            // Maybe handle Solana return data.
60                            |_| { #maybe_return }
61                        )
62                    }
63                }
64            };
65
66            cpi_method
67        })
68        .collect();
69
70    let accounts = generate_accounts(program);
71
72    quote! {
73        #[cfg(feature = "cpi")]
74        pub mod cpi {
75            use super::*;
76            use std::marker::PhantomData;
77
78
79            pub struct Return<T> {
80                phantom: std::marker::PhantomData<T>
81            }
82
83            impl<T: AnchorDeserialize> Return<T> {
84                pub fn get(&self) -> T {
85                    let (_key, data) = anchor_lang::solana_program::program::get_return_data().unwrap();
86                    T::try_from_slice(&data).unwrap()
87                }
88            }
89
90            #(#global_cpi_methods)*
91
92            #accounts
93        }
94    }
95}

97pub fn generate_accounts(program: &Program) -> proc_macro2::TokenStream {
98    let mut accounts = std::collections::HashMap::new();
99
100    // Go through instruction accounts.
101    for ix in &program.ixs {
102        let anchor_ident = &ix.anchor_ident;
103        // TODO: move to fn and share with accounts.rs.
104        let macro_name = format!(
105            "__cpi_client_accounts_{}",
106            anchor_ident.to_string().to_snake_case()
107        );
108        let cfgs = &ix.cfgs;
109        accounts.insert(macro_name, cfgs.as_slice());
110    }
111
112    // Build the tokens from all accounts
113    let account_structs: Vec<proc_macro2::TokenStream> = accounts
114        .iter()
115        .map(|(macro_name, cfgs)| {
116            let macro_name: proc_macro2::TokenStream = macro_name.parse().unwrap();
117            quote! {
118                #(#cfgs)*
119                pub use crate::#macro_name::*;
120            }
121        })
122        .collect();
123
124    quote! {
125        /// An Anchor generated module, providing a set of structs
126        /// mirroring the structs deriving `Accounts`, where each field is
127        /// an `AccountInfo`. This is useful for CPI.
128        pub mod accounts {
129            #(#account_structs)*
130        }
131    }
132}
