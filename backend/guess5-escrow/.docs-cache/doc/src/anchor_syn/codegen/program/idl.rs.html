112        pub trait IdlTrailingData<'info> {
113            fn trailing_data(self) -> Ref<'info, [u8]>;
114            fn trailing_data_mut(self) -> RefMut<'info, [u8]>;

117        impl<'a, 'info: 'a> IdlTrailingData<'a> for &'a Account<'info, IdlAccount> {
118            fn trailing_data(self) -> Ref<'a, [u8]> {
119                let info: &AccountInfo<'info> = self.as_ref();
120                Ref::map(info.try_borrow_data().unwrap(), |d| &d[44..])
121            }
122            fn trailing_data_mut(self) -> RefMut<'a, [u8]> {
123                let info: &AccountInfo<'info> = self.as_ref();
124                RefMut::map(info.try_borrow_mut_data().unwrap(), |d| &mut d[44..])
125            }

129        // One time IDL account initializer. Will fail on subsequent
130        // invocations.
131        #[inline(never)]
132        pub fn __idl_create_account(
133            program_id: &Pubkey,
134            accounts: &mut IdlCreateAccounts,
135            data_len: u64,
136        ) -> anchor_lang::Result<()> {
137            #[cfg(not(feature = "no-log-ix-name"))]
138            anchor_lang::prelude::msg!("Instruction: IdlCreateAccount");
139
140            if program_id != accounts.program.key {
141                return Err(anchor_lang::error::ErrorCode::IdlInstructionInvalidProgram.into());
142            }
143            // Create the IDL's account.
144            let from = accounts.from.key;
145            let (base, nonce) = Pubkey::find_program_address(&[], program_id);
146            let seed = IdlAccount::seed();
147            let owner = accounts.program.key;
148            let to = Pubkey::create_with_seed(&base, seed, owner).unwrap();
149            // Space: account discriminator || authority pubkey || vec len || vec data
150            let space = std::cmp::min(
151                IdlAccount::DISCRIMINATOR.len() + 32 + 4 + data_len as usize,
152                10_000
153            );
154            let rent = Rent::get()?;
155            let lamports = rent.minimum_balance(space);
156            let seeds = &[&[nonce][..]];
157            let ix = anchor_lang::solana_program::system_instruction::create_account_with_seed(
158                from,
159                &to,
160                &base,
161                seed,
162                lamports,
163                space as u64,
164                owner,
165            );
166            anchor_lang::solana_program::program::invoke_signed(
167                &ix,
168                &[
169                    accounts.from.clone(),
170                    accounts.to.clone(),
171                    accounts.base.clone(),
172                    accounts.system_program.to_account_info(),
173                ],
174                &[seeds],
175            )?;
176
177            // Deserialize the newly created account.
178            let mut idl_account = {
179                let mut account_data =  accounts.to.try_borrow_data()?;
180                let mut account_data_slice: &[u8] = &account_data;
181                IdlAccount::try_deserialize_unchecked(
182                    &mut account_data_slice,
183                )?
184            };
185
186            // Set the authority.
187            idl_account.authority = *accounts.from.key;
188
189            // Store the new account data.
190            let mut data = accounts.to.try_borrow_mut_data()?;
191            let dst: &mut [u8] = &mut data;
192            let mut cursor = std::io::Cursor::new(dst);
193            idl_account.try_serialize(&mut cursor)?;
194
195            Ok(())
196        }

198        #[inline(never)]
199        pub fn __idl_resize_account(
200            program_id: &Pubkey,
201            accounts: &mut IdlResizeAccount,
202            data_len: u64,
203        ) -> anchor_lang::Result<()> {
204            #[cfg(not(feature = "no-log-ix-name"))]
205            anchor_lang::prelude::msg!("Instruction: IdlResizeAccount");
206
207            let data_len: usize = data_len as usize;
208
209            // We're not going to support increasing the size of accounts that already contain data
210            // because that would be messy and possibly dangerous
211            if accounts.idl.data_len != 0 {
212                return Err(anchor_lang::error::ErrorCode::IdlAccountNotEmpty.into());
213            }
214
215            let idl_ref = AsRef::<AccountInfo>::as_ref(&accounts.idl);
216            let new_account_space = idl_ref.data_len().checked_add(std::cmp::min(
217                data_len
218                    .checked_sub(idl_ref.data_len())
219                    .expect("data_len should always be >= the current account space"),
220                10_000,
221            ))
222            .unwrap();
223
224            if new_account_space > idl_ref.data_len() {
225                let sysvar_rent = Rent::get()?;
226                let new_rent_minimum = sysvar_rent.minimum_balance(new_account_space);
227                anchor_lang::system_program::transfer(
228                    anchor_lang::context::CpiContext::new(
229                        accounts.system_program.to_account_info(),
230                        anchor_lang::system_program::Transfer {
231                            from: accounts.authority.to_account_info(),
232                            to: accounts.idl.to_account_info(),
233                        },
234                    ),
235                    new_rent_minimum
236                        .checked_sub(idl_ref.lamports())
237                        .unwrap(),
238                )?;
239                idl_ref.realloc(new_account_space, false)?;
240            }
241
242            Ok(())
243
244        }

270        #[inline(never)]
271        pub fn __idl_write(
272            program_id: &Pubkey,
273            accounts: &mut IdlAccounts,
274            idl_data: Vec<u8>,
275        ) -> anchor_lang::Result<()> {
276            #[cfg(not(feature = "no-log-ix-name"))]
277            anchor_lang::prelude::msg!("Instruction: IdlWrite");
278
279            let prev_len: usize = ::std::convert::TryInto::<usize>::try_into(accounts.idl.data_len).unwrap();
280            let new_len: usize = prev_len.checked_add(idl_data.len()).unwrap() as usize;
281            accounts.idl.data_len = accounts.idl.data_len.checked_add(::std::convert::TryInto::<u32>::try_into(idl_data.len()).unwrap()).unwrap();
282
283            use IdlTrailingData;
284            let mut idl_bytes = accounts.idl.trailing_data_mut();
285            let idl_expansion = &mut idl_bytes[prev_len..new_len];
286            require_eq!(idl_expansion.len(), idl_data.len());
287            idl_expansion.copy_from_slice(&idl_data[..]);
288
289            Ok(())
290        }

318            let source = &accounts.buffer.trailing_data()[..buffer_len];
319            require_gte!(target.len(), buffer_len);
320            target[..buffer_len].copy_from_slice(source);
