7pub fn struct_de(input: &ItemStruct, cratename: Ident) -> syn::Result<TokenStream2> {
8    let name = &input.ident;
9    let (impl_generics, ty_generics, where_clause) = input.generics.split_for_impl();
10    let mut where_clause = where_clause.map_or_else(
11        || WhereClause {
12            where_token: Default::default(),
13            predicates: Default::default(),
14        },
15        Clone::clone,
16    );
17    let init_method = contains_initialize_with(&input.attrs)?;
18    let return_value = match &input.fields {
19        Fields::Named(fields) => {
20            let mut body = TokenStream2::new();
21            for field in &fields.named {
22                let field_name = field.ident.as_ref().unwrap();
23                let delta = if contains_skip(&field.attrs) {
24                    quote! {
25                        #field_name: Default::default(),
26                    }
27                } else {
28                    let field_type = &field.ty;
29                    where_clause.predicates.push(
30                        syn::parse2(quote! {
31                            #field_type: #cratename::BorshDeserialize
32                        })
33                        .unwrap(),
34                    );
35
36                    quote! {
37                        #field_name: #cratename::BorshDeserialize::deserialize_reader(reader)?,
38                    }
39                };
40                body.extend(delta);
41            }
42            quote! {
43                Self { #body }
44            }
45        }
46        Fields::Unnamed(fields) => {
47            let mut body = TokenStream2::new();
48            for _ in 0..fields.unnamed.len() {
49                let delta = quote! {
50                    #cratename::BorshDeserialize::deserialize_reader(reader)?,
51                };
52                body.extend(delta);
53            }
54            quote! {
55                Self( #body )
56            }
57        }
58        Fields::Unit => {
59            quote! {
60                Self {}
61            }
62        }
63    };
64    if let Some(method_ident) = init_method {
65        Ok(quote! {
66            impl #impl_generics #cratename::de::BorshDeserialize for #name #ty_generics #where_clause {
67                fn deserialize_reader<R: borsh::maybestd::io::Read>(reader: &mut R) -> ::core::result::Result<Self, #cratename::maybestd::io::Error> {
68                    let mut return_value = #return_value;
69                    return_value.#method_ident();
70                    Ok(return_value)
71                }
72            }
73        })
74    } else {
75        Ok(quote! {
76            impl #impl_generics #cratename::de::BorshDeserialize for #name #ty_generics #where_clause {
77                fn deserialize_reader<R: borsh::maybestd::io::Read>(reader: &mut R) -> ::core::result::Result<Self, #cratename::maybestd::io::Error> {
78                    Ok(#return_value)
79                }
80            }
81        })
82    }
83}
