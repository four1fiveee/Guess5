152fn in_leading_0s_scalar<const BYTE_CNT: usize>(bytes: *const u8) -> u64 {
153    let mut in_leading_0s = 0;
154    while in_leading_0s < BYTE_CNT {
155        if unsafe { bytes.add(in_leading_0s).read() != 0 } {
156            break;
157        }
158        in_leading_0s += 1;
159    }
160    in_leading_0s as u64
161}

182    intermediate: &mut Intermediate<INTERMEDIATE_SZ_W_PADDING>,

194#[inline(always)]
195fn intermediate_to_base58_scalar<
196    const INTERMEDIATE_SZ_W_PADDING: usize,
197    const RAW58_SZ: usize,
198    const INTERMEDIATE_SZ: usize,
199>(
200    intermediate: &Intermediate<INTERMEDIATE_SZ_W_PADDING>,
201    in_leading_0s: u64,
202    out: &mut [u8],
203) -> usize {
204    /* Convert intermediate form to base 58.  This form of conversion
205    exposes tons of ILP, but it's more than the CPU can take advantage
206    of.
207      X = sum_i raw_base58[i] * 58^(RAW58_SZ-1-i) */
208    let mut raw_base58 = [0u8; RAW58_SZ];
209    for i in 0..INTERMEDIATE_SZ {
210        /* We know intermediate[ i ] < 58^5 < 2^32 for all i, so casting to
211        a uint is safe.  GCC doesn't seem to be able to realize this, so
212        when it converts ulong/ulong to a magic multiplication, it
213        generates the single-op 64b x 64b -> 128b mul instruction.  This
214        hurts the CPU's ability to take advantage of the ILP here. */
215        let v = unsafe { *intermediate.0.get_unchecked(i) } as u32;
216        unsafe {
217            *raw_base58.get_unchecked_mut(5 * i + 4) = (v % 58) as u8;
218        }
219        unsafe {
220            *raw_base58.get_unchecked_mut(5 * i + 3) = ((v / 58) % 58) as u8;
221        }
222        unsafe {
223            *raw_base58.get_unchecked_mut(5 * i + 2) = ((v / 3364) % 58) as u8;
224        }
225        unsafe {
226            *raw_base58.get_unchecked_mut(5 * i + 1) = ((v / 195112) % 58) as u8;
227        }
228        unsafe {
229            *raw_base58.get_unchecked_mut(5 * i) = (v / 11316496) as u8;
230        } /* We know this one is less than 58 */
231    }
232    /* Finally, actually convert to the string.  We have to ignore all the
233    leading zeros in raw_base58 and instead insert in_leading_0s
234    leading '1' characters.  We can show that raw_base58 actually has
235    at least in_leading_0s, so we'll do this by skipping the first few
236    leading zeros in raw_base58. */
237    let mut raw_leading_0s = 0;
238    while raw_leading_0s < RAW58_SZ {
239        if raw_base58[raw_leading_0s] != 0 {
240            break;
241        }
242        raw_leading_0s += 1;
243    }
244    /* It's not immediately obvious that raw_leading_0s >= in_leading_0s,
245    but it's true.  In base b, X has floor(log_b X)+1 digits.  That
246    means in_leading_0s = N-1-floor(log_256 X) and raw_leading_0s =
247    RAW58_SZ-1-floor(log_58 X).  Let X<256^N be given and consider:
248
249    raw_leading_0s - in_leading_0s =
250      =  RAW58_SZ-N + floor( log_256 X ) - floor( log_58 X )
251      >= RAW58_SZ-N - 1 + ( log_256 X - log_58 X ) .
252
253    log_256 X - log_58 X is monotonically decreasing for X>0, so it
254    achieves it minimum at the maximum possible value for X, i.e.
255    256^N-1.
256      >= RAW58_SZ-N-1 + log_256(256^N-1) - log_58(256^N-1)
257
258    When N==32, RAW58_SZ is 45, so this gives skip >= 0.29
259    When N==64, RAW58_SZ is 90, so this gives skip >= 1.59.
260
261    Regardless, raw_leading_0s - in_leading_0s >= 0. */
262    let skip = raw_leading_0s - in_leading_0s as usize;
263    for i in 0..(RAW58_SZ - skip) {
264        unsafe {
265            *out.get_unchecked_mut(i) = BASE58_CHARS[raw_base58[skip + i] as usize];
266        }
267    }
268    skip
269}

293#[inline(always)]
294fn adjust_intermediate_array<
295    const INTERMEDIATE_SZ_W_PADDING: usize,
296    const INTERMEDIATE_SZ: usize,
297>(
298    intermediate: &mut Intermediate<INTERMEDIATE_SZ_W_PADDING>,
299) {
300    /* Now we make sure each term is less than 58^5. Again, we have to be
301    a bit careful of overflow.
302
303    For N==32, in the worst case, as before, intermediate[8] will be
304    just over 2^63 and intermediate[7] will be just over 2^62.6.  In
305    the first step, we'll add floor(intermediate[8]/58^5) to
306    intermediate[7].  58^5 is pretty big though, so intermediate[7]
307    barely budges, and this is still fine.
308
309    For N==64, in the worst case, the biggest entry in intermediate at
310    this point is 2^63.87, and in the worst case, we add (2^64-1)/58^5,
311    which is still about 2^63.87. */
312    for i in (1..=INTERMEDIATE_SZ - 1).rev() {
313        unsafe {
314            *intermediate.0.get_unchecked_mut(i - 1) += *intermediate.0.get_unchecked(i) / R1DIV;
315        }
316        unsafe {
317            *intermediate.0.get_unchecked_mut(i) %= R1DIV;
318        }
319    }
320}

339#[cfg(not(target_feature = "avx2"))]
340#[inline(always)]
341fn u8s_to_u32s_scalar<const N: usize, const BINARY_SZ: usize>(
342    out: &mut [u8; N],
343    binary_u8: &[u8; N],
344) {
345    for i in 0..BINARY_SZ {
346        let idx = i * 4;
347        #[cfg(target_endian = "little")]
348        unsafe {
349            *out.get_unchecked_mut(idx) = *binary_u8.get_unchecked(idx + 3);
350            *out.get_unchecked_mut(idx + 1) = *binary_u8.get_unchecked(idx + 2);
351            *out.get_unchecked_mut(idx + 2) = *binary_u8.get_unchecked(idx + 1);
352            *out.get_unchecked_mut(idx + 3) = *binary_u8.get_unchecked(idx);
353        }
354        #[cfg(target_endian = "big")]
355        unsafe {
356            *out.get_unchecked_mut(idx) = *binary_u8.get_unchecked(idx);
357            *out.get_unchecked_mut(idx + 1) = *binary_u8.get_unchecked(idx + 1);
358            *out.get_unchecked_mut(idx + 2) = *binary_u8.get_unchecked(idx + 2);
359            *out.get_unchecked_mut(idx + 3) = *binary_u8.get_unchecked(idx + 3);
360        }
361    }
362}

371#[cfg(target_feature = "avx2")]
372#[inline(always)]
373fn u8s_to_u32s_swapped_64_register(bytes: &[u8; N_64]) -> [__m256i; 2] {
374    let mask = u8s_to_u32s_swapped_mask_32();
375    let first_load =
376        unsafe { _mm256_loadu_si256(bytes.get_unchecked(..32).as_ptr() as *const __m256i) };
377    let first = unsafe { _mm256_shuffle_epi8(first_load, mask) };
378    [first, unsafe {
379        _mm256_shuffle_epi8(
380            _mm256_loadu_si256(bytes.get_unchecked(32..64).as_ptr() as *const __m256i),
381            mask,
382        )
383    }]
384}

420fn make_binary_array_32(bytes: &[u8; N_32]) -> [u32; BINARY_SZ_32] {
421    // on LE take four-byte blocks and reverse them
422    // 3 2 1 0 7 6 5 4 etc
423    // on BE just take four-byte blocks
424
425    let mut out = [0u8; N_32];
426    #[cfg(target_feature = "avx2")]
427    {
428        u8s_to_u32s_swapped_32(bytes, &mut out);
429    }
430    #[cfg(not(target_feature = "avx2"))]
431    {
432        u8s_to_u32s_scalar::<N_32, BINARY_SZ_32>(&mut out, bytes);
433    }
434    unsafe { core::mem::transmute(out) }
435}

465/// Encode a 64-byte array.
466///
467/// Mutates the provided `out` array and returns a u8 `len`
468/// which indicates how many bytes of the `out` array were actually written to.
469/// The remaining bytes are unchanged, hence the result we care about after
470/// calling the function is `out[..len as usize]`.
471///
472/// # Examples
473/// ```
474/// let mut buf = [0u8; 88];
475/// let bytes = &[
476///     0, 0, 10, 85, 198, 191, 71, 18, 5, 54, 6, 255, 181, 32, 227, 150, 208, 3, 157, 135, 222,
477///     67, 50, 23, 237, 51, 240, 123, 34, 148, 111, 84, 98, 162, 236, 133, 31, 93, 185, 142, 108,
478///     41, 191, 1, 138, 6, 192, 0, 46, 93, 25, 65, 243, 223, 225, 225, 85, 55, 82, 251, 109, 132,
479///     165, 2,
480/// ];
481/// let len = five8::encode_64(bytes, &mut buf);
482/// assert_eq!(
483///     &buf[..len as usize],
484///     [
485///         49, 49, 99, 103, 84, 72, 52, 68, 53, 101, 56, 83, 51, 115, 110, 68, 52, 52, 52, 87, 98,
486///         98, 71, 114, 107, 101, 112, 106, 84, 118, 87, 77, 106, 50, 106, 107, 109, 67, 71, 74, 116,
487///         103, 110, 51, 72, 55, 113, 114, 80, 98, 49, 66, 110, 119, 97, 112, 120, 112, 98, 71, 100,
488///         82, 116, 72, 81, 104, 57, 116, 57, 87, 98, 110, 57, 116, 54, 90, 68, 71, 72, 122, 87, 112,
489///         76, 52, 100, 102
490///     ]
491/// );
492/// assert_eq!(len, 86);
493#[inline]
494pub fn encode_64(bytes: &[u8; N_64], out: &mut [u8; BASE58_ENCODED_64_MAX_LEN]) -> u8 {
495    let bytes_ptr = bytes as *const u8;
496    let in_leading_0s = {
497        #[cfg(target_feature = "avx2")]
498        {
499            in_leading_0s_64_avx(bytes_ptr)
500        }
501        #[cfg(not(target_feature = "avx2"))]
502        {
503            in_leading_0s_scalar::<N_64>(bytes_ptr)
504        }
505    };
506    let binary = make_binary_array_64(bytes);
507    let intermediate = make_intermediate_array_64(binary);
508    let skip = {
509        #[cfg(not(target_feature = "avx2"))]
510        {
511            intermediate_to_base58_scalar::<
512                INTERMEDIATE_SZ_W_PADDING_64,
513                RAW58_SZ_64,
514                INTERMEDIATE_SZ_64,
515            >(&intermediate, in_leading_0s, out)
516        }
517        #[cfg(target_feature = "avx2")]
518        {
519            let intermediate_ptr = intermediate.0.as_ptr() as *const i64;
520            let raw0 = intermediate_to_raw(wl_ld(intermediate_ptr));
521            let raw1 = intermediate_to_raw(wl_ld(unsafe { intermediate_ptr.offset(4) }));
522            let raw2 = intermediate_to_raw(wl_ld(unsafe { intermediate_ptr.offset(8) }));
523            let raw3 = intermediate_to_raw(wl_ld(unsafe { intermediate_ptr.offset(12) }));
524            let raw4 = intermediate_to_raw(wl_ld(unsafe { intermediate_ptr.offset(16) }));
525            let (compact0, compact1, compact2) = ten_per_slot_down_64(raw0, raw1, raw2, raw3, raw4);
526            let raw_leading_0s_part1 = count_leading_zeros_64(compact0, compact1);
527            let raw_leading_0s_part2 = count_leading_zeros_26(compact2);
528            let raw_leading_0s = if raw_leading_0s_part1 < 64 {
529                raw_leading_0s_part1
530            } else {
531                64 + raw_leading_0s_part2
532            };
533            let base58_0 = raw_to_base58(compact0);
534            let base58_1 = raw_to_base58(compact1);
535            let base58_2 = raw_to_base58(compact2);
536            let skip = raw_leading_0s - in_leading_0s;
537            /* We'll do something similar.  The final string is between 64 and 88
538            characters, so skip is [2, 26].
539            */
540            let w_skip = wl_bcast(skip as i64);
541            let mod8_mask = wl_bcast(7);
542            let compare = wl(0, 1, 2, 3);
543            let shift_qty = wl_shl::<3>(wl_and(w_skip, mod8_mask)); /* bytes->bits */
544            let shifted = wl_shru_vector(base58_0, shift_qty);
545            let skip_div8 = wl_shru::<3>(w_skip);
546            let mask1 = wl_eq(skip_div8, compare);
547            let mask2 = wl_gt(compare, skip_div8);
548            let out_ptr = out.as_mut_ptr();
549            unsafe {
550                _mm256_maskstore_epi64(
551                    (out_ptr.offset(-8 * (skip as isize / 8))) as *mut i64,
552                    mask1,
553                    shifted,
554                )
555            };
556            unsafe {
557                _mm256_maskstore_epi64(
558                    out_ptr.offset(-(skip as isize)) as *mut i64,
559                    mask2,
560                    base58_0,
561                )
562            };
563
564            unsafe { wuc_stu(out_ptr.offset(32 - skip as isize), base58_1) };
565
566            let last = unsafe { _mm_bslli_si128(_mm256_extractf128_si256(base58_2, 1), 6) };
567            unsafe {
568                _mm_storeu_si128(
569                    out_ptr.offset(64 + 16 - 6 - skip as isize) as *mut __m128i,
570                    last,
571                )
572            };
573            unsafe {
574                _mm_storeu_si128(
575                    out_ptr.offset(64 - skip as isize) as *mut __m128i,
576                    _mm256_extractf128_si256(base58_2, 0),
577                )
578            };
579            skip
580        }
581    };
582    RAW58_SZ_64 as u8 - skip as u8
583}
584
585#[inline(always)]
586fn make_intermediate_array_64(
587    binary: [u32; BINARY_SZ_64],
588) -> Intermediate<INTERMEDIATE_SZ_W_PADDING_64> {
589    /* Convert to the intermediate format:
590      X = sum_i intermediate[i] * 58^(5*(INTERMEDIATE_SZ-1-i))
591    Initially, we don't require intermediate[i] < 58^5, but we do want
592    to make sure the sums don't overflow. */
593    let mut intermediate = init_intermediate_array::<INTERMEDIATE_SZ_W_PADDING_64>();
594
595    /* If we do it the same way as the 32B conversion, intermediate[16]
596    can overflow when the input is sufficiently large.  We'll do a
597    mini-reduction after the first 8 steps.  After the first 8 terms,
598    the largest intermediate[16] can be is 2^63.87.  Then, after
599    reduction it'll be at most 58^5, and after adding the last terms,
600    it won't exceed 2^63.1.  We do need to be cautious that the
601    mini-reduction doesn't cause overflow in intermediate[15] though.
602    Pre-mini-reduction, it's at most 2^63.05.  The mini-reduction adds
603    at most 2^64/58^5, which is negligible.  With the final terms, it
604    won't exceed 2^63.69, which is fine. Other terms are less than
605    2^63.76, so no problems there. */
606    for i in 0..8 {
607        for j in 0..INTERMEDIATE_SZ_64 - 1 {
608            let multiplier = unsafe { *ENC_TABLE_64.get_unchecked(i).get_unchecked(j) } as u64;
609            add_binary_to_intermediate(&mut intermediate, j, binary, i, multiplier);
610        }
611    }
612    /* Mini-reduction */
613    unsafe {
614        *intermediate.0.get_unchecked_mut(15) += intermediate.0.get_unchecked(16) / R1DIV;
615    }
616    unsafe {
617        *intermediate.0.get_unchecked_mut(16) %= R1DIV;
618    }
619    /* Finish iterations */
620    for i in 8..BINARY_SZ_64 {
621        for j in 0..INTERMEDIATE_SZ_64 - 1 {
622            let multiplier = unsafe { *ENC_TABLE_64.get_unchecked(i).get_unchecked(j) as u64 };
623            add_binary_to_intermediate(&mut intermediate, j, binary, i, multiplier);
624        }
625    }
626    adjust_intermediate_array::<INTERMEDIATE_SZ_W_PADDING_64, INTERMEDIATE_SZ_64>(
627        &mut intermediate,
628    );
629    intermediate
630}

646#[cfg(target_feature = "avx2")]
647#[inline(always)]
648fn intermediate_to_base58_32_avx(
649    intermediate: &Intermediate<INTERMEDIATE_SZ_W_PADDING_32>,
650    in_leading_0s: u64,
651    out: &mut [u8],
652) -> usize {
653    let intermediate_ptr = intermediate.0.as_ptr() as *const i64;
654    let intermediate0 = wl_ld(intermediate_ptr);
655    let intermediate1 = wl_ld(unsafe { intermediate_ptr.offset(4) });
656    let intermediate2 = wl_ld(unsafe { intermediate_ptr.offset(8) });
657    let raw0 = intermediate_to_raw(intermediate0);
658    let raw1 = intermediate_to_raw(intermediate1);
659    let raw2 = intermediate_to_raw(intermediate2);
660    let (compact0, compact1) = ten_per_slot_down_32(raw0, raw1, raw2);
661    let raw_leading_0s = count_leading_zeros_45(compact0, compact1);
662    let base58_0 = raw_to_base58(compact0);
663    let base58_1 = raw_to_base58(compact1);
664    let skip = raw_leading_0s as usize - in_leading_0s as usize;
665    /* We know the final string is between 32 and 44 characters, so skip
666     has to be in [1, 13].
667
668     Suppose base58_0 is [ a0 a1 a2 ... af | b0 b1 b2 ... bf ] and skip
669     is 2.
670     It would be nice if we had something like the 128-bit barrel shifts
671     we used in ten_per_slot_down, but they require immediates.
672     Instead, we'll shift each ulong right by (skip%8) bytes:
673
674     [ a2 a3 .. a7 0 0 aa ab .. af 0 0 | b2 b3 .. b7 0 0 ba .. bf 0 0 ]
675     and maskstore to write just 8 bytes, skipping the first
676     floor(skip/8) ulongs, to a starting address of out-8*floor(skip/8).
677
678           out
679             V
680           [ a2 a3 a4 a5 a6 a7  0  0 -- -- ... ]
681
682     Now we use another maskstore on the original base58_0, masking out
683     the first floor(skip/8)+1 ulongs, and writing to out-skip:
684           out
685             V
686     [ -- -- -- -- -- -- -- -- a8 a9 aa ab ... bd be bf ]
687
688     Finally, we need to write the low 13 bytes from base58_1 and a '\0'
689     terminator, starting at out-skip+32.  The easiest way to do this is
690     actually to shift in 3 more bytes, write the full 16B and do it
691     prior to the previous write:
692                                               out-skip+29
693                                                V
694                                              [ 0  0  0  c0 c1 c2 .. cc ]
695    */
696    let w_skip = wl_bcast(skip as i64);
697    let mod8_mask = wl_bcast(7);
698    let compare = wl(0, 1, 2, 3);
699    let shift_qty = wl_shl::<3>(wl_and(w_skip, mod8_mask)); /* bytes->bits */
700    let shifted = wl_shru_vector(base58_0, shift_qty);
701    let skip_div8 = wl_shru::<3>(w_skip);
702    let mask1 = wl_eq(skip_div8, compare);
703    let out_ptr = out.as_mut_ptr();
704    let out_offset = unsafe { out_ptr.offset(-8 * (skip as isize / 8)) } as *mut i64;
705    unsafe { _mm256_maskstore_epi64(out_offset, mask1, shifted) };
706    let last = unsafe { _mm_bslli_si128(_mm256_extractf128_si256(base58_1, 0), 3) };
707    unsafe { _mm_storeu_si128(out_ptr.offset(29 - skip as isize) as *mut __m128i, last) };
708    let mask2 = wl_gt(compare, skip_div8);
709    unsafe {
710        _mm256_maskstore_epi64(
711            out_ptr.offset(-(skip as isize)) as *mut i64,
712            mask2,
713            base58_0,
714        )
715    };
716    skip
717}

719/// Encode a 32-byte array.
720///
721/// Mutates the provided `out` array and returns a u8 `len`
722/// which indicates how many bytes of the `out` array were actually written to.
723/// The remaining bytes are unchanged, hence the result we care about after
724/// calling the function is `out[..len as usize]`.
725///
726/// # Examples
727/// ```
728/// let mut buf = [0u8; 44];
729/// let bytes = &[
730///     24, 243, 6, 223, 230, 153, 210, 8, 92, 137, 123, 67, 164, 197, 79, 196, 125, 43, 183,
731///     85, 103, 91, 232, 167, 73, 131, 104, 131, 0, 101, 214, 231,
732/// ];
733/// let len = five8::encode_32(bytes, &mut buf);
734/// assert_eq!(
735///     &buf[..len as usize],
736///     [
737///         50, 103, 80, 105, 104, 85, 84, 106, 116, 51, 70, 74, 113, 102, 49, 86, 112, 105,
738///         100, 103, 114, 89, 53, 99, 90, 54, 80, 117, 121, 77, 99, 99, 71, 86, 119, 81, 72,
739///         82, 102, 106, 77, 80, 90, 71
740///     ]
741/// );
742/// assert_eq!(len, 44);
743/// ```
744#[inline]
745pub fn encode_32(bytes: &[u8; N_32], out: &mut [u8; BASE58_ENCODED_32_MAX_LEN]) -> u8 {
746    let bytes_ptr = bytes as *const u8;
747    let in_leading_0s = {
748        #[cfg(target_feature = "avx2")]
749        {
750            in_leading_0s_32_avx(bytes_ptr)
751        }
752        #[cfg(not(target_feature = "avx2"))]
753        {
754            in_leading_0s_scalar::<N_32>(bytes_ptr)
755        }
756    };
757    let binary = make_binary_array_32(bytes);
758    let intermediate = make_intermediate_array_32(binary);
759    let skip = intermediate_to_base58_32(&intermediate, in_leading_0s, out);
760    RAW58_SZ_32 as u8 - skip as u8
761}

763#[inline(always)]
764fn intermediate_to_base58_32(
765    intermediate: &Intermediate<INTERMEDIATE_SZ_W_PADDING_32>,
766    in_leading_0s: u64,
767    out: &mut [u8],
768) -> usize {
769    #[cfg(not(target_feature = "avx2"))]
770    {
771        intermediate_to_base58_scalar::<INTERMEDIATE_SZ_W_PADDING_32, RAW58_SZ_32, INTERMEDIATE_SZ_32>(
772            intermediate,
773            in_leading_0s,
774            out,
775        )
776    }
777    #[cfg(target_feature = "avx2")]
778    intermediate_to_base58_32_avx(intermediate, in_leading_0s, out)
779}
