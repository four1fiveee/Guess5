119#[derive(Debug, PartialEq)]
120pub enum DecodeError {
121    InvalidChar(u8),
122    TooLong,
123    TooShort,
124    LargestTermTooHigh,
125    OutputTooLong,
126}
127
128#[cfg(feature = "std")]
129impl std::error::Error for DecodeError {}
130
131#[cfg(feature = "std")]
132impl fmt::Display for DecodeError {
133    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
134        match self {
135            DecodeError::InvalidChar(c) => {
136                write!(f, "Illegal base58 char number: {}", c)
137            }
138            DecodeError::TooLong => f.write_str("Base58 string too long"),
139            DecodeError::TooShort => f.write_str("Base58 string too short"),
140            DecodeError::LargestTermTooHigh => f.write_str("Largest term greater than 2^32"),
141            DecodeError::OutputTooLong => f.write_str("Decoded output has too many bytes"),
142        }
143    }
144}
